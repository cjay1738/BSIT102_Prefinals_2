import java.util.ArrayList;
import java.util.Scanner;
import java.util.Base64;
import java.io.IOException;
import java.io.File; // Import the File class
import java.io.FileWriter;
import java.io.Writer;

public class Encrypt
{
    public static void main(String[] args) throws IOException
    {
    	ArrayList<String> word = new ArrayList<>(); // Create an Arraylist Files
    	Scanner scr = new Scanner(System.in);
    	Boolean isDone = false;
    	
    	while(!isDone) // Code block to be executed
    	{
    		String wordIN = scr.next();
    		if(!wordIN.equals(":q"))  // Block of code to be executed if the condition is true
    		{
    			byte[] encodedBytes = Base64.getEncoder().encode((wordIN + "\n").getBytes()); // Encode the bytes to Base64
    			word.add(new String(encodedBytes));
    		}
    		else
    		{
    			isDone = true; // Outputs true
    		}
    	}
    	
    	FileWriter fwr = new FileWriter("Encrypted.cjam"); 
    	for(String a: word)  // Code block to be executed
    	{
    		fwr.write(a);    
    	}
    	
      	fwr.close();  // The statement will stop

    	
    	 
       	
    }
}






import java.io.IOException;

import java.util.Base64;

public class Decrypt {
    public static void main(String[] args) throws IOException{
    	
    	Scanner scr = new Scanner(new File("Encrypted.cjam"));

    	byte[] decodedBytes = Base64.getDecoder().decode(scr.next()); // Decode the Base64 String to bytes
		System.out.println(new String(decodedBytes));
    	
    }
}
